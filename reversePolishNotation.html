From now on, I have a plan. One day, one algorithm question.

Flies are also meat.

First one

Evaluate the value of an arithmetic expression in Reverse Polish Notation. Valid operators are +, -, *, /. Each operand may be an integer or another expression.

Some examples:

["2", "1", "+", "3", "*"] -> ((2 + 1) * 3) -> 9
["4", "13", "5", "/", "+"] -> (4 + (13 / 5)) -> 6

algorithm idea:

Use the stack data structure. Traverse the string array from the first to the end, if the element is operand, then push it into the stack; if the element is operator, then pop out two elements from the stack and compute them with the operator, the result will also push into the stack. At last, the only element of the stack is the value of the reverse polish notation

The following is the source code:

/**
 * Evaluate the value of an arithmetic expression in Reverse Polish Notation.
 * Valid operators are +, -, *, /. Each operand may be an integer or another expression.
 * Some examples:
 * ["2", "1", "+", "3", "*"] -> ((2 + 1) * 3) -> 9
 * ["4", "13", "5", "/", "+"] -> (4 + (13 / 5)) -> 6
 * */
package sort;

public class ComputeReversePolishNotation {

    public static void main(String[] args) {

        String[] s = { "2", "1", "+", "3", "*" };
        String[] s1 = { "4", "13", "5", "/", "+" };
        System.out.println(new ComputeReversePolishNotation()
                .computeReversePolishNotation(s));
        System.out.println(new ComputeReversePolishNotation()
                .computeReversePolishNotation(s1));

    }

    /**
     * Compute reverse polish notation
     * 
     * @param stack
     *            s stack which store the reverse polish notation
     * @return result
     * */
    public int computeReversePolishNotation(String[] s) {
        if (s == null) {
            return 0;
        }

        MyStack myStack = new MyStack(null);

        for (int i = 0; i < s.length; i++) {
            if (isOperator(s[i])) {
                String operand1 = myStack.pop();
                String operand2 = myStack.pop();
                int temp = computeTwoNumbersWithAOperator(operand2, operand1,
                        s[i]);// reverse the operand
                myStack.push(String.valueOf(temp));
            } else {
                myStack.push(s[i]);
            }
        }

        if (myStack.isEmpty()) {
            return 0;
        } else {
            return Integer.valueOf(myStack.pop());// it should be only one
                                                    // element in the stack
        }

    }

    /**
     * judge whether the string is a operator
     * */
    private boolean isOperator(String s) {
        if ("+".equals(s) || "-".equals(s) || "*".equals(s) || "/".equals(s)) {
            return true;
        } else {
            return false;
        }
    }

    /**
     * simple compute
     * */
    private int computeTwoNumbersWithAOperator(String operand1,
            String operand2, String operator) {
        int op1 = Integer.valueOf(operand1);
        int op2 = Integer.valueOf(operand2);

        if ("+".equals(operator)) {
            return (op1 + op2);
        } else if ("-".equals(operator)) {
            return (op1 - op2);
        } else if ("*".equals(operator)) {
            return (op1 * op2);
        } else if ("/".equals(operator)) {
            return (op1 / op2);
        } else {
            return 0;
        }

    }

}

package sort;

import java.util.ArrayList;

public class MyStack {

    private ArrayList<String> list = new ArrayList<String>();

    /**
     * constructor
     * */
    public MyStack(String[] s) {
        if (s == null) {
            return;
        }

        for(int i = 0; i < s.length; i++) {
            list.add(s[i]);
        }
    }

    /**
     * push
     * */
    public void push(String s) {
        list.add(s);
    }

    /**
     * pop
     * */
    public String pop() {
        if (isEmpty()) {
            return null;
        }

        String temp = list.get(list.size()-1);
        list.remove(list.size()-1);
        return temp;
    }

    /**
     * judge whether the stack is empty or not
     * */
    public boolean isEmpty() {
        if (list.size() == 0) {
            return true;
        } else {
            return false;
        }
    }
